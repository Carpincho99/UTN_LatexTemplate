INICIO;
int operacion;
float a, b, resultado;
Leer(operacion);
Si (operacion == 1)
{
    Leer(a, b);
    resultado = (a + b);
}
Sino Si (operacion == 2)
{
    Leer(a, b);
    resultado = (a * b);
}
Sino Si (operacion == 3)
{
    Leer(a, b);
    resultado = (a - b);
}
Sino Si (operacion == 4)
{
    Leer(a, b);
    resultado = (a / b);
}
else
{
    Imp("Error, $operacion no es una operacion valida.");
    FIN; 
}
Imp("Resultado de la operacion: $resultado");
